def self.hsv_to_rgb(h, s, v)
  h = [359, [h, 0].max].min.fdiv(360)
  s = s.fdiv(100)
  v = v.fdiv(100)

  h_i = (h * 6).to_i

  f = h * 6 - h_i
  p = v * (1 - s)
  q = v * (1 - f * s)
  t = v * (1 - (1 - f) * s)

  case h_i
  when 0
    r = v
    g = t
    b = p
  when 1
    r = q
    g = v
    b = p
  when 2
    r = p
    g = v
    b = t
  when 3
    r = p
    g = q
    b = v
  when 4
    r = t
    g = p
    b = v
  when 5
    r = v
    g = p
    b = q
  end
  [(r * 255).to_i, (g * 255).to_i, (b * 255).to_i]
end

    a[0].send(op, b[0]),
    a[1].send(op, b[1])
  ]
end
