def truncate(width)
  if @max_width <= width
    @text = @full_text
    return
  end

  (@full_text.length - 1).downto 0 do |i|
    truncated = "#{@full_text[0..i]}#{ellipsis}"
    cur_width, = $gtk.calcstringbox(truncated, @size, font)
    if cur_width <= width
      @text = truncated
      return # rubocop:disable Lint/NonLocalExitFromIterator
    end
  end

  @text = ''
end
